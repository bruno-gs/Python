Inverter uma frase:
frase = str(input(“Digite uma frase: ”))
palavras = frase.split		#Separa a frase numa lista
junto = ‘’.joint(palavras)	#Junta tudo sem espaços, formando uma palavra gigante
inverso =’’
for letra in range(len(junto)-1, -1, -1):
	inverso += junto[letra]
print(junto, inverso)
